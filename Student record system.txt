class Student:
    def _init_(self, roll, name, marks):
        self.roll = roll
        self.name = name
        self.marks = marks
        self.next = None

class StudentManager:
    def _init_(self):
        self.head = None

    def add_student(self):
        roll = int(input("Enter Roll No: "))
        name = input("Enter Name: ")
        marks = float(input("Enter Marks: "))
        new_student = Student(roll, name, marks)
        new_student.next = self.head
        self.head = new_student
        print("Student added.\n")

    def display_all(self):
        temp = self.head
        if not temp:
            print("No records found.\n")
            return
        while temp:
            print(f"Roll: {temp.roll}, Name: {temp.name}, Marks: {temp.marks}")
            temp = temp.next
        print()

    def search_student(self):
        roll = int(input("Enter Roll No to search: "))
        temp = self.head
        while temp:
            if temp.roll == roll:
                print(f"Found: {temp.name} with marks {temp.marks}\n")
                return
            temp = temp.next
        print("Student not found.\n")

    def delete_student(self):
        roll = int(input("Enter Roll No to delete: "))
        temp = self.head
        prev = None
        while temp:
            if temp.roll == roll:
                if prev:
                    prev.next = temp.next
                else:
                    self.head = temp.next
                print("Student deleted.\n")
                return
            prev = temp
            temp = temp.next
        print("Student not found.\n")

def main():
    manager = StudentManager()
    while True:
        print("1. Add\n2. Display\n3. Search\n4. Delete\n5. Exit")
        choice = input("Choice: ")
        if choice == '1':
            manager.add_student()
        elif choice == '2':
            manager.display_all()
        elif choice == '3':
            manager.search_student()
        elif choice == '4':
            manager.delete_student()
        elif choice == '5':
            break
        else:
            print("Invalid choice.\n")

if _name_ == "_main_":
    main()
